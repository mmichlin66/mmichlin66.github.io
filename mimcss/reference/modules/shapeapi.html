    // filter: blur(2px)
    blurred = this.$class({ filter: css.blur(2)})

    // filter: blur(1.5em)

    Returns IBlurFunc

Const dropShadow

  • dropShadow(x: Extended<CssLength>, y: Extended<CssLength>, color?: Extended<CssColor>, blur?: Extended<CssLength>): IDropShadowFunc

  • Returns an IDropShadowFunc object representing the dropShadow() CSS function parameters.

    Example

    class MyStyles extends css.StyleDefinition
{
    // filter: drop-shadow(30px 10px 4px blue)
    blurredShadow = this.$class({ filter: css.dropShadow( 30, 10, "blue", 4)})

    // filter: drop-shadow(2.5em -1.5em green)
    sharpShadow = this.$class({ filter: css.dropShadow( 2.5, -1.5, "green")})

    // filter: drop-shadow(0 0 20px orange)
    haloShadow = this.$class({ filter: css.dropShadow( 0, 0, "orange", 20px)})
}

    Parameters

    • x: Extended<CssLength>

      Horizontal offset of the shadow.

    • y: Extended<CssLength>

      Vertical offset of the shadow.

    • Optional color: Extended<CssColor>

      Color of the shadow. If undefined, the color of the shadow is taken from the color property.

    • Optional blur: Extended<CssLength>

      Value of the shadow's blurring. If undefined, the shadow will be sharp (not blurred).

    Returns IDropShadowFunc

    The IDropShadowFunc interface containing the shadow parameters.
